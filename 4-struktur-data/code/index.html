# List 

# Operasi pada List di Python

# 1. Membuat List
# Membuat berbagai jenis list: list string, list angka, dan list campuran.
buah_buahan = ["apel", "pisang", "jeruk"]
angka = [1, 2, 3, 4, 5]
campuran = ["halo", 10, True]

print("List Buah:", buah_buahan)       # Output: List Buah: ['apel', 'pisang', 'jeruk']
print("List Angka:", angka)            # Output: List Angka: [1, 2, 3, 4, 5]
print("List Campuran:", campuran)      # Output: List Campuran: ['halo', 10, True]
print()

# 2. Mengakses Elemen di List
# Mengakses elemen menggunakan indeks positif dan negatif.
print("Elemen pertama buah_buahan:", buah_buahan[0])    # Output: apel
print("Elemen kedua buah_buahan:", buah_buahan[1])      # Output: pisang
print("Elemen terakhir buah_buahan:", buah_buahan[-1])  # Output: jeruk
print()

# 3. Mengubah Elemen di List
# Mengubah nilai elemen tertentu dalam list.
print("List buah sebelum diubah:", buah_buahan)          # Output: ['apel', 'pisang', 'jeruk']
buah_buahan[1] = "mangga"
print("List buah setelah diubah:", buah_buahan)          # Output: ['apel', 'mangga', 'jeruk']
print()

# 4. Menambah Elemen ke List
# a. Menggunakan append() untuk menambah elemen di akhir list.
print("List buah sebelum append:", buah_buahan)          # Output: ['apel', 'mangga', 'jeruk']
buah_buahan.append("anggur")
print("List buah setelah append:", buah_buahan)          # Output: ['apel', 'mangga', 'jeruk', 'anggur']
print()

# b. Menggunakan insert() untuk menyisipkan elemen di posisi tertentu.
print("List buah sebelum insert:", buah_buahan)          # Output: ['apel', 'mangga', 'jeruk', 'anggur']
buah_buahan.insert(1, "durian")
print("List buah setelah insert:", buah_buahan)          # Output: ['apel', 'durian', 'mangga', 'jeruk', 'anggur']
print()

# c. Menggunakan extend() untuk menggabungkan dua list.
sayuran = ["wortel", "bayam"]
print("List buah sebelum extend:", buah_buahan)           # Output: ['apel', 'durian', 'mangga', 'jeruk', 'anggur']
buah_buahan.extend(sayuran)
print("List buah setelah extend:", buah_buahan)           # Output: ['apel', 'durian', 'mangga', 'jeruk', 'anggur', 'wortel', 'bayam']
print()

# 5. Menghapus Elemen dari List
# a. Menggunakan remove() untuk menghapus elemen berdasarkan nilai.
print("List buah sebelum remove:", buah_buahan)           # Output: ['apel', 'durian', 'mangga', 'jeruk', 'anggur', 'wortel', 'bayam']
buah_buahan.remove("jeruk")
print("List buah setelah remove:", buah_buahan)           # Output: ['apel', 'durian', 'mangga', 'anggur', 'wortel', 'bayam']
print()

# b. Menggunakan pop() untuk menghapus elemen berdasarkan indeks.
print("List buah sebelum pop:", buah_buahan)              # Output: ['apel', 'durian', 'mangga', 'anggur', 'wortel', 'bayam']
dihapus = buah_buahan.pop(2)                              # Menghapus elemen di indeks 2 ("mangga")
print(f"Elemen yang dihapus: {dihapus}")                  # Output: mangga
print("List buah setelah pop:", buah_buahan)              # Output: ['apel', 'durian', 'anggur', 'wortel', 'bayam']
print()

# c. Menggunakan clear() untuk menghapus semua elemen dalam list.
print("List buah sebelum clear:", buah_buahan)            # Output: ['apel', 'durian', 'anggur', 'wortel', 'bayam']
buah_buahan.clear()
print("List buah setelah clear:", buah_buahan)            # Output: []
print()

# 6. Menggunakan Loop pada List
# Menggunakan for loop untuk mencetak setiap elemen dalam list.
print("Menggunakan loop untuk mencetak elemen list buah_buahan:")
buah_buahan = ["apel", "pisang", "jeruk"]
for buah in buah_buahan:
    print(f"Buah: {buah}")
# Output:
# Buah: apel
# Buah: pisang
# Buah: jeruk
print()

# 7. Fungsi-fungsi Berguna untuk List
# a. Menggunakan len() untuk menghitung jumlah elemen dalam list.
print("Jumlah elemen dalam list buah_buahan:", len(buah_buahan))  # Output: 3

# b. Menggunakan sort() untuk mengurutkan elemen dalam list.
angka = [3, 1, 4, 2]
print("List angka sebelum sort:", angka)          # Output: [3, 1, 4, 2]
angka.sort()
print("List angka setelah sort:", angka)          # Output: [1, 2, 3, 4]

# c. Menggunakan reverse() untuk membalik urutan elemen dalam list.
print("List angka sebelum reverse:", angka)       # Output: [1, 2, 3, 4]
angka.reverse()
print("List angka setelah reverse:", angka)       # Output: [4, 3, 2, 1]

# d. Menggunakan 'in' untuk mengecek keberadaan elemen dalam list.
print("Apakah 'apel' ada dalam list buah_buahan?", "apel" in buah_buahan)  # Output: True
print("Apakah 'mangga' ada dalam list buah_buahan?", "mangga" in buah_buahan)  # Output: False
print()

# 8. List dalam List (Nested List)
# Membuat dan mengakses elemen dalam nested list.
nested_list = [[1, 2, 3], ["apel", "jeruk"]]
print("Elemen pertama dari nested_list:", nested_list[0])          # Output: [1, 2, 3]
print("Elemen kedua dari list kedua dalam nested_list:", nested_list[1][1])  # Output: jeruk

# Tuple 

# Operasi pada Tuple di Python

# 1. Membuat Tuple
# Membuat berbagai jenis tuple: tuple string, tuple angka, dan tuple campuran.
hewan = ("kucing", "anjing", "kelinci")
angka = (1, 2, 3, 4, 5)
campuran = ("halo", 10, True)

print("Tuple Hewan:", hewan)         # Output: Tuple Hewan: ('kucing', 'anjing', 'kelinci')
print("Tuple Angka:", angka)         # Output: Tuple Angka: (1, 2, 3, 4, 5)
print("Tuple Campuran:", campuran)   # Output: Tuple Campuran: ('halo', 10, True)
print()

# 2. Mengakses Elemen di Tuple
# Mengakses elemen menggunakan indeks positif dan negatif.
print("Elemen pertama hewan:", hewan[0])    # Output: kucing
print("Elemen kedua hewan:", hewan[1])      # Output: anjing
print("Elemen terakhir hewan:", hewan[-1])  # Output: kelinci
print()

# 3. Mengubah Elemen di Tuple (Immutable)
# Mencoba mengubah elemen akan menghasilkan error.
print("Tuple sebelum mencoba diubah:", hewan)  # Output: ('kucing', 'anjing', 'kelinci')
try:
    hewan[1] = "burung"  # Ini akan menimbulkan error
except TypeError as e:
    print("Error saat mengubah tuple:", e)       # Output: Error saat mengubah tuple: 'tuple' object does not support item assignment
print()

# 4. Mengubah Tuple dengan Mengonversinya ke List
# Mengonversi tuple ke list, mengubah elemen, lalu kembali ke tuple.
print("Tuple sebelum diubah:", hewan)          # Output: ('kucing', 'anjing', 'kelinci')
hewan_list = list(hewan)                       # Mengonversi ke list
hewan_list[1] = "burung"                       # Mengubah elemen di list
hewan_baru = tuple(hewan_list)                 # Mengonversi kembali ke tuple
print("Tuple setelah diubah:", hewan_baru)     # Output: ('kucing', 'burung', 'kelinci')
print()

# 5. Menambah Elemen ke Tuple (Immutable)
# Menggabungkan tuple yang ada dengan tuple baru untuk menambah elemen.
buah_buahan = ("apel", "mangga", "jeruk")
print("Tuple buah sebelum penambahan:", buah_buahan)  # Output: ('apel', 'mangga', 'jeruk')
buah_baru = ("pisang",)
buah_buahan = buah_buahan + buah_baru
print("Tuple buah setelah penambahan:", buah_buahan)   # Output: ('apel', 'mangga', 'jeruk', 'pisang')
print()

# 6. Unpacking Tuple
# Memisahkan elemen-elemen dalam tuple ke dalam variabel yang berbeda.
print("Unpacking tuple buah_buahan:")
buah1, buah2, buah3, buah4 = buah_buahan
print("Buah 1:", buah1)  # Output: apel
print("Buah 2:", buah2)  # Output: mangga
print("Buah 3:", buah3)  # Output: jeruk
print("Buah 4:", buah4)  # Output: pisang
print()

# 7. Fungsi-fungsi Tuple
# a. Menggunakan len() untuk menghitung jumlah elemen dalam tuple.
print("Jumlah elemen dalam tuple buah_buahan:", len(buah_buahan))  # Output: 4

# b. Menggunakan count() untuk menghitung berapa kali elemen tertentu muncul dalam tuple.
angka_duplikat = (1, 2, 2, 3, 4)
print("Jumlah kemunculan angka 2 dalam angka_duplikat:", angka_duplikat.count(2))  # Output: 2

# c. Menggunakan index() untuk menemukan indeks dari elemen tertentu.
print("Indeks angka 3 dalam angka:", angka.index(3))  # Output: 2
print()

# 8. Tuple dalam Tuple (Nested Tuple)
# Membuat dan mengakses elemen dalam nested tuple.
nested_tuple = ((1, 2, 3), ("apel", "mangga"))
print("Nested Tuple:", nested_tuple)                       # Output: ((1, 2, 3), ('apel', 'mangga'))
print("Elemen pertama dari nested_tuple:", nested_tuple[0])       # Output: (1, 2, 3)
print("Elemen kedua dari nested_tuple:", nested_tuple[1][1])     # Output: mangga
print()

# Dictionary 

# Operasi pada Dictionary di Python

# 1. Membuat Dictionary
# Membuat berbagai jenis dictionary: dictionary string, dictionary angka, dan dictionary campuran.
mahasiswa = {
    "nama": "Budi",
    "umur": 21,
    "jurusan": "Teknik Informatika"
}
print("Dictionary Mahasiswa:", mahasiswa)
# Output: Dictionary Mahasiswa: {'nama': 'Budi', 'umur': 21, 'jurusan': 'Teknik Informatika'}

angka = {
    1: "satu",
    2: "dua",
    3: "tiga"
}
print("Dictionary Angka:", angka)
# Output: Dictionary Angka: {1: 'satu', 2: 'dua', 3: 'tiga'}

campuran = {
    "angka": 10,
    "kata": "halo",
    True: "benar"
}
print("Dictionary Campuran:", campuran)
# Output: Dictionary Campuran: {'angka': 10, 'kata': 'halo', True: 'benar'}
print()

# 2. Mengakses Elemen di Dictionary
# Mengakses elemen menggunakan key.
print("Nama Mahasiswa:", mahasiswa["nama"])    # Output: Budi
print("Umur Mahasiswa:", mahasiswa["umur"])    # Output: 21
print("Jurusan Mahasiswa:", mahasiswa["jurusan"])  # Output: Teknik Informatika
print()

# Mengakses Key yang Tidak Ada (akan menyebabkan KeyError)
# print(mahasiswa["alamat"])  # KeyError: 'alamat'

# Menggunakan Metode get() untuk mengakses elemen yang mungkin tidak ada
print("Alamat Mahasiswa (menggunakan get):", mahasiswa.get("alamat"))  # Output: None
print("Alamat Mahasiswa (dengan default):", mahasiswa.get("alamat", "Tidak diketahui"))  # Output: Tidak diketahui
print()

# 3. Menambah atau Mengubah Elemen di Dictionary
# Menambah elemen baru
print("Dictionary Mahasiswa sebelum penambahan:", mahasiswa)
mahasiswa["alamat"] = "Jl. Mawar No. 10"
print("Dictionary Mahasiswa setelah penambahan:", mahasiswa)
# Output:
# Dictionary Mahasiswa sebelum penambahan: {'nama': 'Budi', 'umur': 21, 'jurusan': 'Teknik Informatika'}
# Dictionary Mahasiswa setelah penambahan: {'nama': 'Budi', 'umur': 21, 'jurusan': 'Teknik Informatika', 'alamat': 'Jl. Mawar No. 10'}

# Mengubah elemen yang sudah ada
print("Dictionary Angka sebelum perubahan:", angka)
angka[2] = "dua besar"
print("Dictionary Angka setelah perubahan:", angka)
# Output:
# Dictionary Angka sebelum perubahan: {1: 'satu', 2: 'dua', 3: 'tiga'}
# Dictionary Angka setelah perubahan: {1: 'satu', 2: 'dua besar', 3: 'tiga'}
print()

# 4. Menghapus Elemen dari Dictionary
# a. Menggunakan pop() untuk menghapus elemen berdasarkan key dan mengembalikan value-nya
print("Dictionary Mahasiswa sebelum pop:", mahasiswa)
umur = mahasiswa.pop("umur")
print(f"Elemen yang dihapus (umur): {umur}")
print("Dictionary Mahasiswa setelah pop:", mahasiswa)
# Output:
# Dictionary Mahasiswa sebelum pop: {'nama': 'Budi', 'umur': 21, 'jurusan': 'Teknik Informatika', 'alamat': 'Jl. Mawar No. 10'}
# Elemen yang dihapus (umur): 21
# Dictionary Mahasiswa setelah pop: {'nama': 'Budi', 'jurusan': 'Teknik Informatika', 'alamat': 'Jl. Mawar No. 10'}

# b. Menggunakan del untuk menghapus elemen berdasarkan key
print("Dictionary Angka sebelum del:", angka)
del angka[3]
print("Dictionary Angka setelah del:", angka)
# Output:
# Dictionary Angka sebelum del: {1: 'satu', 2: 'dua besar', 3: 'tiga'}
# Dictionary Angka setelah del: {1: 'satu', 2: 'dua besar'}
print()

# c. Menggunakan clear() untuk menghapus semua elemen dalam dictionary
print("Dictionary Campuran sebelum clear:", campuran)
campuran.clear()
print("Dictionary Campuran setelah clear:", campuran)
# Output:
# Dictionary Campuran sebelum clear: {'angka': 10, 'kata': 'halo', True: 'benar'}
# Dictionary Campuran setelah clear: {}
print()

# 5. Looping di Dictionary
# a. Loop berdasarkan key
print("Looping berdasarkan key:")
for key in mahasiswa:
    print(f"Key: {key}")
# Output:
# Key: nama
# Key: jurusan
# Key: alamat
print()

# b. Loop berdasarkan value
print("Looping berdasarkan value:")
for value in mahasiswa.values():
    print(f"Value: {value}")
# Output:
# Value: Budi
# Value: Teknik Informatika
# Value: Jl. Mawar No. 10
print()

# c. Loop berdasarkan key dan value
print("Looping berdasarkan key dan value:")
for key, value in mahasiswa.items():
    print(f"{key}: {value}")
# Output:
# nama: Budi
# jurusan: Teknik Informatika
# alamat: Jl. Mawar No. 10
print()

# 6. Fungsi-fungsi Berguna untuk Dictionary
# a. Menggunakan len() untuk menghitung jumlah elemen dalam dictionary
print("Jumlah elemen dalam dictionary mahasiswa:", len(mahasiswa))  # Output: 3
print()

# b. Menggunakan keys() untuk mendapatkan semua key dalam dictionary
print("Semua key dalam dictionary mahasiswa:", mahasiswa.keys())
# Output: dict_keys(['nama', 'jurusan', 'alamat'])

# c. Menggunakan values() untuk mendapatkan semua value dalam dictionary
print("Semua value dalam dictionary mahasiswa:", mahasiswa.values())
# Output: dict_values(['Budi', 'Teknik Informatika', 'Jl. Mawar No. 10'])

# d. Menggunakan items() untuk mendapatkan semua pasangan key-value dalam dictionary
print("Semua pasangan key-value dalam dictionary mahasiswa:", mahasiswa.items())
# Output: dict_items([('nama', 'Budi'), ('jurusan', 'Teknik Informatika'), ('alamat', 'Jl. Mawar No. 10')])
print()

# e. Menggunakan update() untuk menambahkan atau mengubah elemen dari dictionary lain
data_baru = {"umur": 22, "hobi": "Membaca"}

mahasiswa.update(data_baru)
print("Dictionary mahasiswa setelah update:", mahasiswa)

# Dictionary mahasiswa setelah update: {'nama': 'Budi', 'jurusan': 'Teknik Informatika', 'alamat': 'Jl. Mawar No. 10', 'umur': 22, 'hobi': 'Membaca'}

buah = {"apel", "jeruk", "mangga"}

campuran = {"halo", 10, True}

# Contoh Lengkap List Comprehension di Python

# 1. Pengantar List Comprehension
# List Comprehension adalah cara singkat dan elegan untuk membuat list baru dari list yang sudah ada atau dari iterable lainnya seperti range, string, dll.

# Sintaks Dasar:
# [expression for item in iterable]

# Contoh Sederhana:
angka = [i for i in range(5)]
print("List angka:", angka)  # Output: [0, 1, 2, 3, 4]

# **Penjelasan:**
# - range(5) menghasilkan angka dari 0 hingga 4.
# - Untuk setiap i dalam range tersebut, i dimasukkan ke dalam list baru 'angka'.
# - Hasilnya adalah list [0, 1, 2, 3, 4].

print("\n" + "="*50 + "\n")

# 2. List Comprehension dengan Operasi Matematika
# Kamu bisa melakukan operasi matematika atau manipulasi data lainnya di dalam list comprehension.

kuadrat = [i**2 for i in range(5)]
print("List kuadrat:", kuadrat)  # Output: [0, 1, 4, 9, 16]

# **Penjelasan:**
# - Untuk setiap i dalam range(5), i pangkat 2 (i**2) dihitung dan dimasukkan ke dalam list 'kuadrat'.
# - Hasilnya adalah list [0, 1, 4, 9, 16].

print("\n" + "="*50 + "\n")

# 3. List Comprehension dengan Kondisi (If)

# **Penjelasan:**
# - Hanya angka yang habis dibagi 2 (genap) yang akan dimasukkan ke dalam list 'genap'.
# - Hasilnya adalah list [0, 2, 4, 6, 8].

print("\n" + "="*50 + "\n")

# Kamu bisa menggunakan if-else untuk memilih nilai yang akan dimasukkan berdasarkan kondisi tertentu.

angka_label = ["genap" if i % 2 == 0 else "ganjil" for i in range(5)]

# Kamu bisa menggunakan nested loops dalam list comprehension untuk membuat kombinasi dari dua atau lebih iterables.

print("\n" + "="*50 + "\n")
