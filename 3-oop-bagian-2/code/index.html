# Contoh Kode

# Inheritance 

class Hewan:
    def __init__(self, nama, umur):
        self.nama = nama    # Public member
        self.umur = umur    # Public member

    def suara(self):
        print("Hewan ini mengeluarkan suara.")

class Anjing(Hewan):
    def __init__(self, nama, umur, breed):
        super().__init__(nama, umur)  # Memanggil konstruktor kelas induk
        self.breed = breed            # Atribut tambahan spesifik untuk Anjing

    def suara(self):
        print(f"{self.nama} menggonggong: Woof Woof!")

class Kucing(Hewan):
    def __init__(self, nama, umur, warna):
        super().__init__(nama, umur)  # Memanggil konstruktor kelas induk
        self.warna = warna            # Atribut tambahan spesifik untuk Kucing

    def suara(self):
        print(f"{self.nama} mengeong: Meow Meow!")

# Membuat objek Anjing
anjing1 = Anjing("Buddy", 5, "Golden Retriever")
anjing1.suara()  # Output: Buddy menggonggong: Woof Woof!
print(f"Nama Hewan: {anjing1.nama}, Umur: {anjing1.umur} tahun, Breed: {anjing1.breed}.")
# Output: Nama Hewan: Buddy, Umur: 5 tahun, Breed: Golden Retriever.

# Membuat objek Kucing
kucing1 = Kucing("Whiskers", 3, "Hitam")
kucing1.suara()  # Output: Whiskers mengeong: Meow Meow!
print(f"Nama Hewan: {kucing1.nama}, Umur: {kucing1.umur} tahun, Warna: {kucing1.warna}.")
# Output: Nama Hewan: Whiskers, Umur: 3 tahun, Warna: Hitam.

# Menggunakan isinstance
print(isinstance(anjing1, Anjing))    # Output: True
print(isinstance(anjing1, Hewan))     # Output: True
print(isinstance(kucing1, Kucing))    # Output: True
print(isinstance(kucing1, Hewan))     # Output: True

# Menggunakan issubclass
print(issubclass(Anjing, Hewan))      # Output: True
print(issubclass(Kucing, Hewan))      # Output: True
print(issubclass(Hewan, Anjing))      # Output: False

# Encapsulation 

class BankAccount:
    def __init__(self, owner, balance):
        self.owner = owner            # Public member
        self._balance = balance        # Protected member
        self.__account_number = "987654321"  # Private member

    def deposit(self, amount):
        if amount > 0:
            self._balance += amount
            print(f"Anda telah menyetor Rp{amount}. Saldo saat ini: Rp{self._balance}.")
        else:
            print("Jumlah setoran harus lebih dari nol.")

    def withdraw(self, amount):
        if 0 < amount <= self._balance:
            self._balance -= amount
            print(f"Anda telah menarik Rp{amount}. Saldo saat ini: Rp{self._balance}.")
        else:
            print("Jumlah penarikan tidak valid.")

    def get_account_number(self):
        return self.__account_number  # Getter untuk atribut private

    def get_balance(self):
        return self._balance  # Getter untuk atribut protected

    def set_balance(self, balance):
        if balance >= 0:
            self._balance = balance  # Setter untuk atribut protected
        else:
            print("Saldo tidak dapat bernilai negatif.")

# Membuat objek BankAccount
akun1 = BankAccount("Budi", 1500000)

# Mengakses public member
print(f"Pemilik akun: {akun1.owner}")  # Output: Pemilik akun: Budi

# Menggunakan metode deposit dan withdraw
akun1.deposit(500000)   # Output: Anda telah menyetor Rp500000. Saldo saat ini: Rp2000000.
akun1.withdraw(300000)  # Output: Anda telah menarik Rp300000. Saldo saat ini: Rp1700000.

# Mengakses protected member melalui getter
print(f"Saldo saat ini: Rp{akun1.get_balance()}")  # Output: Saldo saat ini: Rp1700000

# Mengakses private member melalui getter
print(f"Nomor Akun: {akun1.get_account_number()}")  # Output: Nomor Akun: 987654321

# Mengubah saldo menggunakan setter
akun1.set_balance(2500000)
print(f"Saldo setelah diubah: Rp{akun1.get_balance()}")  # Output: Saldo setelah diubah: Rp2500000

# Mencoba mengakses protected dan private member langsung (tidak disarankan)
print(f"Saldo langsung (tidak disarankan): Rp{akun1._balance}")  # Output: Saldo langsung (tidak disarankan): Rp2500000

# Polymorphism 

# Langkah 1: Mendefinisikan kelas Kucing dan Anjing
class Kucing:
    def suara(self):
        print("Kucing mengeong: Meow!")

class Anjing:
    def suara(self):
        print("Anjing menggonggong: Guk!")

# Langkah 2: Membuat objek dari kelas Kucing dan Anjing
kucing = Kucing()
anjing = Anjing()

# Langkah 3: Memasukkan objek ke dalam daftar dan menggunakan loop
hewan = [kucing, anjing]

for h in hewan:
    h.suara()

# Langkah 4: Menambahkan kelas Burung
class Burung:
    def suara(self):
        print("Burung berkicau: Cuit cuit!")

# Memasukkan Burung ke dalam daftar hewan dan memanggil metode suara()
burung = Burung()
hewan.append(burung)

for h in hewan:
    h.suara()

# Abstraction 

from abc import ABC, abstractmethod

# Langkah 1: Membuat kelas abstrak Kendaraan
class Kendaraan(ABC):
    @abstractmethod
    def bergerak(self):
        pass
    
    @abstractmethod
    def bahan_bakar(self):
        pass

# Langkah 2: Membuat kelas Mobil yang mewarisi Kendaraan
class Mobil(Kendaraan):
    def bergerak(self):
        print("Mobil bergerak di jalan.")
    
    def bahan_bakar(self):
        print("Mobil menggunakan bensin.")

# Langkah 3: Membuat kelas Sepeda yang mewarisi Kendaraan
class Sepeda(Kendaraan):
    def bergerak(self):
        print("Sepeda bergerak dengan pedal.")
    
    def bahan_bakar(self):
        print("Sepeda tidak menggunakan bahan bakar.")

# Langkah 4: Membuat objek dari subkelas dan memanggil metode
mobil = Mobil()
sepeda = Sepeda()

mobil.bergerak()        # Output: Mobil bergerak di jalan.
mobil.bahan_bakar()     # Output: Mobil menggunakan bensin.

sepeda.bergerak()       # Output: Sepeda bergerak dengan pedal.
sepeda.bahan_bakar()    # Output: Sepeda tidak menggunakan bahan bakar.
